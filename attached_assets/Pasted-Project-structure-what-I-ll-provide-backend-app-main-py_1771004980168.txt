Backend (FastAPI)

File: backend/app/main.py
```python
from fastapi import FastAPI, File, UploadFile
from fastapi.responses import JSONResponse, FileResponse
import uvicorn
import tempfile
import os
import subprocess
import librosa
import numpy as np
import soundfile as sf

app = FastAPI()

def analyze_bpm_key(filepath):
y, sr = librosa.load(filepath, sr=None, mono=True)
# BPM
tempo, beats = librosa.beat.beat_track(y=y, sr=sr)
# Key estimation (simple: chroma & find max)
chroma = librosa.feature.chroma_cens(y=y, sr=sr)
chroma_mean = chroma.mean(axis=1)
key_idx = int(np.argmax(chroma_mean))
# Map index to note name (C, C#/Db, D, ...)
notes = ['C','C#','D','D#','E','F','F#','G','G#','A','A#','B']
key = notes[key_idx]
return {"bpm": float(tempo), "key": key}

@app.post("/analyze")
async def analyze(file: UploadFile = File(...)):
suffix = os.path.splitext(file.filename)[1] or ".wav"
with tempfile.NamedTemporaryFile(delete=False, suffix=suffix) as tmp:
content = await file.read()
tmp.write(content)
tmp.flush()
tmp_path = tmp.name
try:
result = analyze_bpm_key(tmp_path)
return JSONResponse(content=result)
except Exception as e:
return JSONResponse(content={"error": str(e)}, status_code=500)
finally:
try:
os.unlink(tmp_path)
except: pass

@app.post("/separate")
async def separate(file: UploadFile = File(...)):
"""
Uses demucs to separate stems. Requires demucs installed in backend environment.
Returns a ZIP of stems or path to stems. For simplicity returns JSON with status and file paths.
"""
suffix = os.path.splitext(file.filename)[1] or ".wav"
with tempfile.TemporaryDirectory() as td:
in_path = os.path.join(td, "input"+suffix)
with open(in_path, "wb") as f:
f.write(await file.read())
out_dir = os.path.join(td, "separated")
os.makedirs(out_dir, exist_ok=True)
# Attempt to run demucs (ensure demucs is in PATH)
cmd = ["demucs", "--two-stems=vocals", "-o", out_dir, in_path]
try:
subprocess.check_output(cmd, stderr=subprocess.STDOUT)
# Find produced files
produced = []
for root,_,files in os.walk(out_dir):
for fn in files:
if fn.endswith(".wav") or fn.endswith(".mp3"):
produced.append(os.path.join(root,fn))
return JSONResponse(content={"status":"done","files":produced})
except subprocess.CalledProcessError as e:
return JSONResponse(content={"error":"demucs failed","output": e.output.decode()}, status_code=500)
except FileNotFoundError:
return JSONResponse(content={"error":"demucs not installed on server"}, status_code=400)

if __name__ == "__main__":
uvicorn.run("app.main:app", host="0.0.0.0", port=8000, reload=True)
```

Quick run instructions (local dev)

1) Backend
- Create virtualenv, install requirements:
```
cd backend
python3 -m venv venv
source venv/bin/activate
pip install -r app/requirements.txt
# Optional: pip install demucs (if you plan to use /separate)
uvicorn app.main:app --reload --port 8000
```
- FastAPI will run at http://localhost:8000. The `/analyze` endpoint is ready.

2) Frontend
- Serve static files:
```
cd frontend
python3 -m http.server 8080
```
- Open http://localhost:8080 in a browser. Upload two local audio files; use Play/Pause, crossfader, auto-mix, record, and Analyze (calls backend at /analyze — ensure CORS or proxy configured if serving frontend and backend separately).
- For CORS: if you serve frontend on a different port, add CORS to backend:
```
pip install "fastapi[all]" fastapi
```

```
from fastapi.middleware.cors import CORSMiddleware
```

backend/app/main.py
```python
from fastapi import FastAPI, File, UploadFile
from fastapi.responses import JSONResponse
from fastapi.middleware.cors import CORSMiddleware
import tempfile, os, subprocess, json
import numpy as np
import librosa

app = FastAPI()
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"])

def analyze_bpm_key(path):
# Attempt several tempo detectors and ensemble
y, sr = librosa.load(path, sr=None, mono=True)
# librosa beat
tempo1, _ = librosa.beat.beat_track(y=y, sr=sr)
# autocorrelation via onset envelope
onset_env = librosa.onset.onset_strength(y=y, sr=sr)
tempo2 = librosa.beat.tempo(onset_envelope=onset_env, sr=sr)
# choose median
tempos = [float(tempo1)] + [float(t) for t in np.atleast_1d(tempo2)]
tempo = float(np.median(tempos))
# key estimate via chroma
chroma = librosa.feature.chroma_cens(y=y, sr=sr)
key_idx = int(np.argmax(chroma.mean(axis=1)))
notes = ['C','C#','D','D#','E','F','F#','G','G#','A','A#','B']
return {"bpm": round(tempo,2), "key": notes[key_idx]}

@app.post("/analyze")
async def analyze(file: UploadFile = File(...)):
suffix = os.path.splitext(file.filename)[1] or ".wav"
with tempfile.NamedTemporaryFile(delete=False, suffix=suffix) as tmp:
content = await file.read()
tmp.write(content)
tmp.flush()
tmp_path = tmp.name
try:
res = analyze_bpm_key(tmp_path)
return JSONResponse(content=res)
finally:
try: os.unlink(tmp_path)
except: pass

@app.post("/separate")
async def separate(file: UploadFile = File(...), two_stems: bool = True):
"""
Runs demucs (if installed) to separate stems.
Returns JSON with paths (on server) or error.
For production return signed URLs to downloaded files from object storage.
"""
suffix = os.path.splitext(file.filename)[1] or ".wav"
with tempfile.TemporaryDirectory() as td:
in_path = os.path.join(td, "input"+suffix)
with open(in_path, "wb") as f:
f.write(await file.read())
out_dir = os.path.join(td, "out")
os.makedirs(out_dir, exist_ok=True)
# demucs command: demucs --two-stems=vocals -o out in_path
cmd = ["demucs", "--two-stems=vocals", "-o", out_dir, in_path] if two_stems else ["demucs", "-o", out_dir, in_path]
try:
subprocess.check_output(cmd, stderr=subprocess.STDOUT)
except FileNotFoundError:
return JSONResponse(content={"error":"demucs not installed on server"}, status_code=400)
except subprocess.CalledProcessError as e:
return JSONResponse(content={"error":"demucs failed","output": e.output.decode()}, status_code=500)
# collect produced wavs
produced = []
for root,_,files in os.walk(out_dir):
for fn in files:
if fn.endswith(".wav") or fn.endswith(".mp3"):
produced.append(os.path.join(root,fn))
return JSONResponse(content={"status":"done","files":produced})

@app.post("/mix-suggestion")
async def mix_suggestion(fileA: UploadFile = File(...), fileB: UploadFile = File(...)):
"""
Analyze both tracks and return a simple mix plan:
- bpm/key of each
- suggested crossfade length (secs)
- suggested BPM rate multiplier for deckB to match deckA
- suggested transition type (fade, echo, filter)
"""
def save_temp(f):
suffix = os.path.splitext(f.filename)[1] or ".wav"
tf = tempfile.NamedTemporaryFile(delete=False, suffix=suffix)
content = f.file.read()
tf.write(content); tf.flush(); tf.close()
return tf.name
a_path = save_temp(fileA); b_path = save_temp(fileB)
try:
a = analyze_bpm_key(a_path)
b = analyze_bpm_key(b_path)
# conservative crossfade: if BPM close, shorter crossfade
bpm_diff = abs(a['bpm'] - b['bpm'])
cross = 6 if bpm_diff < 3 else 10 if bpm_diff < 6 else 14
# suggested rate to match B -> A
rate = round(a['bpm'] / b['bpm'], 4) if b['bpm']>0 else 1.0
# simple transition type heuristic
if a['key'] == b['key']:
ttype = "blend"
elif bpm_diff < 2:
ttype = "filter-sweep"
else:
ttype = "beat-jump"
return JSONResponse(content={"A":a, "B":b, "suggested_rate_for_B":rate, "crossfade_seconds":cross, "transition":ttype})
finally:
try: os.unlink(a_path); os.unlink(b_path)
except: pass

if __name__ == "__main__":
import uvicorn
```

```javascript
const log = document.getElementById('log');
let mapping = {}; // user-defined mapping: midiNote -> action

function append(s){ log.textContent += s + '\n'; }

if(navigator.requestMIDIAccess){
navigator.requestMIDIAccess().then(midi => {
append('MIDI ready');
for(let input of midi.inputs.values()){
input.onmidimessage = e => {
const [status, data1, data2] = e.data;
append(`msg: status=${status} data1=${data1} data2=${data2}`);
// Example: if control held, allow mapping to an action
// Simple UI-less demo: if velocity>0 set mapping for note to toggle playA
if(status === 144 && data2 > 0){ // NoteOn
if(!mapping[data1]){
mapping[data1] = 'playA';
append(`Mapped note ${data1} -> playA`);
} else {
// trigger action
const action = mapping[data1];
append(`Trigger action ${action} for note ${data1}`);
// In real integration you'd call playDeck('A') etc.
}
}
};
}
});
} else append('WebMIDI not supported');
```
